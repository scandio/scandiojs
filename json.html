   var
      t = typeof (obj),
      n,
      v,
      json = [],
      arr = null;

   if (t != "object" || obj === null) {

      return String(obj);
   } else {
      arr = (obj && obj.constructor == Array);

      for (n in obj) {
         v = obj[n];
         t = typeof(v);
         if (obj.hasOwnProperty(n)) {
            if (t == "string") {
               v = '"' + v + '"';
            } else if (t == "object" && v !== null){
               v = jQuery.stringify(v);
            }

            json.push((arr ? "" : '"' + n + '":') + String(v));
         }
      }

      return (arr ? "[" : "{") + String(json) + (arr ? "]" : "}");
   }
